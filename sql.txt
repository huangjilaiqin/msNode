CREATE DATABASE IF NOT EXISTS mstemplate DEFAULT CHARSET utf8 COLLATE utf8_general_ci;
use mstemplate;

GRANT ALL PRIVILEGES ON mstemplate.* TO 'mstemplate'@'localhost' IDENTIFIED BY 'mstemplate_pass' WITH GRANT OPTION;

CREATE TABLE `articles` (
	`id` INT(11) NOT NULL AUTO_INCREMENT,
	`title` VARCHAR(50) NULL DEFAULT '0',
	`listImg` VARCHAR(50) NULL DEFAULT '0',
	`summary` VARCHAR(50) NULL DEFAULT '0',
	`content` VARCHAR(50) NULL DEFAULT '0',
	`authorId` INT(11) NOT NULL DEFAULT '0',
	`tags` VARCHAR(50) NOT NULL DEFAULT '0',
	`createdAt` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
	`updatedAt` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
	PRIMARY KEY (`id`)
)
COLLATE='utf8_general_ci'
ENGINE=InnoDB
AUTO_INCREMENT=2
;



CREATE TABLE `comments` (
	`id` INT(11) NOT NULL AUTO_INCREMENT,
	`articleId` INT(11) NOT NULL DEFAULT '0',
	`content` VARCHAR(50) NULL DEFAULT NULL,
	`commentator` INT(11) NULL DEFAULT NULL,
	`becommentator` INT(11) NULL DEFAULT NULL,
	`createdAt` DATETIME NULL DEFAULT CURRENT_TIMESTAMP,
	`updatedAt` DATETIME NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
	PRIMARY KEY (`id`)
)
COLLATE='utf8_general_ci'
ENGINE=InnoDB
AUTO_INCREMENT=2
;




CREATE TABLE `users` (
	`id` INT(11) NOT NULL AUTO_INCREMENT,
	`nickname` VARCHAR(255) NULL DEFAULT NULL,
	`introduce` VARCHAR(255) NULL DEFAULT '',
	`birthday` DATETIME NULL DEFAULT NULL,
	`createdAt` DATETIME NULL DEFAULT CURRENT_TIMESTAMP,
	`updatedAt` DATETIME NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
	PRIMARY KEY (`id`)
)
COLLATE='utf8_general_ci'
ENGINE=InnoDB
AUTO_INCREMENT=9
;

CREATE TABLE `sys_users` (
	`id` INT(11) NOT NULL AUTO_INCREMENT,
	`userName` VARCHAR(50) NOT NULL DEFAULT '0',
	`password` VARCHAR(50) NOT NULL DEFAULT '0',
	`createdAt` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
	`updatedAt` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
	PRIMARY KEY (`id`)
)
COLLATE='utf8_general_ci'
ENGINE=InnoDB
;
